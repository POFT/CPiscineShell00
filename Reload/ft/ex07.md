Solution:

1. Header (F1)
2. Define Library (#include <unistd.h>)
3. Code:

ft_print_numbers.c

        void    ft_print_numbers(void);
        void    ft_putchar(char c);
        
        void    ft_print_numbers(void)
        {
                char    number;
        
                number = '0';
                while (number <= '9')
                {
                        ft_putchar(number);
                        number++;
                }
        }
        
        int     main(void)
        {
                ft_print_numbers();
                return (0);
        }
